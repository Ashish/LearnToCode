//**LINKED LIST IN JAVA USING ARRAY**//

public class LinkedList 
{
	Node head;
	
	public void insert(int data)
	{
		Node node = new Node();
		node.data = data;
		node.next = null;
		if(head == null)
		{
			head = node;
		} 
		else
		{
			Node n = head;
			while(n.next != null)
			{
				n  = n.next;
			}
			n.next = node;
		}
	}
	
	public void show()
	{
		Node currNode = head;
		while(currNode != null)
		{
			System.out.print(currNode.data + " ");
			currNode = currNode.next;
		}
	}

}
